SELECT nombrepila,
  COUNT(EMPLEADO_TENER_SUCURSAL.IDEMPLEADO) as Numero_Sucursales
FROM EMPLEADO_TENER_SUCURSAL
JOIN EMPLEADO
ON(EMPLEADO_TENER_SUCURSAL.idempleado = EMPLEADO.idempleado)
GROUP BY nombrepila, EMPLEADO_TENER_SUCURSAL.idempleado;

  SELECT IDSUCURSAL, IDPRODUCTO, NOMBRE
  FROM PRODUCTO_ABARROTE
  JOIN SUCURSAL_TENER_ABARROTES
  ON (PRODUCTO_ABARROTE.IDDEPARTAMENTO = SUCURSAL_TENER_ABARROTES.IDDEPARTAMENTO)
  ORDER BY 1 ASC;

  SELECT IDSUCURSAL, IDPRODUCTO, NOMBRE
  FROM PRODUCTO_MED_CONTROLADO
  JOIN SUCURSAL_TENER_FARMACIA
  ON (PRODUCTO_MED_CONTROLADO.IDDEPARTAMENTO = SUCURSAL_TENER_FARMACIA.IDDEPARTAMENTO)
  ORDER BY 1 ASC;

  SELECT IDSUCURSAL, IDPRODUCTO, NOMBRE
  FROM PRODUCTO_MED_LIBRE
  JOIN SUCURSAL_TENER_FARMACIA
  ON (PRODUCTO_MED_LIBRE.IDDEPARTAMENTO = SUCURSAL_TENER_FARMACIA.IDDEPARTAMENTO)
  ORDER BY 1 ASC;

  SELECT IDSUCURSAL, IDPRODUCTO, NOMBRE
  FROM PRODUCTO_VINO
  JOIN SUCURSAL_TENER_VINOSYLICORES
  ON (PRODUCTO_VINO.IDDEPARTAMENTO = SUCURSAL_TENER_VINOSYLICORES.IDDEPARTAMENTO)
  ORDER BY 1 ASC;

  SELECT SUCURSAL.IDSUCURSAL, SUCURSAL_TENER_ABARROTES.IDDEPARTAMENTO AS ABARROTES, SUCURSAL_TENER_FARMACIA.IDDEPARTAMENTO AS FARMACIA, SUCURSAL_TENER_VINOSYLICORES.IDDEPARTAMENTO AS VINOS
  FROM SUCURSAL
  INNER JOIN SUCURSAL_TENER_ABARROTES
  ON (SUCURSAL.IDSUCURSAL = SUCURSAL_TENER_ABARROTES.IDSUCURSAL)
  INNER JOIN SUCURSAL_TENER_FARMACIA
  ON (SUCURSAL.IDSUCURSAL = SUCURSAL_TENER_FARMACIA.IDSUCURSAL)
  INNER JOIN SUCURSAL_TENER_VINOSYLICORES
  ON (SUCURSAL.IDSUCURSAL = SUCURSAL_TENER_VINOSYLICORES.IDSUCURSAL);

SELECT SUCURSAL_TENER_ABARROTES.IDSUCURSAL AS Sucursal,
  PRODUCTO_ABARROTE.IDPRODUCTO             AS Departamento ,
  nombre                                   AS PRODUCTO
FROM PRODUCTO_ABARROTE
JOIN SUCURSAL_TENER_ABARROTES
ON (PRODUCTO_ABARROTE.IDDEPARTAMENTO = SUCURSAL_TENER_ABARROTES.IDDEPARTAMENTO)
ORDER BY 1 ASC;

SELECT SUCURSAL_TENER_FARMACIA.IDSUCURSAL AS Sucursal,
  PRODUCTO_MED_CONTROLADO.IDPRODUCTO      AS Departamento ,
  nombre                                  AS PRODUCTO
FROM PRODUCTO_MED_CONTROLADO
JOIN SUCURSAL_TENER_FARMACIA
ON (PRODUCTO_MED_CONTROLADO.IDDEPARTAMENTO = SUCURSAL_TENER_FARMACIA.IDDEPARTAMENTO)
ORDER BY 1 ASC;

SELECT SUCURSAL_TENER_FARMACIA.IDSUCURSAL AS Sucursal,
  PRODUCTO_MED_LIBRE.IDPRODUCTO      AS Departamento ,
  nombre                                  AS PRODUCTO
FROM PRODUCTO_MED_LIBRE
JOIN SUCURSAL_TENER_FARMACIA
ON (PRODUCTO_MED_LIBRE.IDDEPARTAMENTO = SUCURSAL_TENER_FARMACIA.IDDEPARTAMENTO)
ORDER BY 1 ASC;

SELECT SUCURSAL_TENER_VINOSYLICORES.IDSUCURSAL AS Sucursal,
  PRODUCTO_VINO.IDPRODUCTO      AS Departamento ,
  nombre                                  AS PRODUCTO
FROM PRODUCTO_VINO
JOIN SUCURSAL_TENER_VINOSYLICORES
ON (PRODUCTO_VINO.IDDEPARTAMENTO = SUCURSAL_TENER_VINOSYLICORES.IDDEPARTAMENTO)
ORDER BY 1 ASC;
